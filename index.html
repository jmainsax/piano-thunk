  <h1>Virtual Piano for Trichords</h1>

  <button onclick="playTrichord(['C4', 'E4', 'G4'])">Play C Major</button>

  <button onclick="playTrichord(['D4', 'F4', 'A4'])">Play D Minor</button>

  <button onclick="playTrichord(['G3', 'B3', 'D4'])">Play G Major</button>


  <script>

    // Create a Synth instance for the piano sound

    const synth = new Tone.PolySynth(Tone.Synth).toDestination();


    // Function to play a trichord

    function playTrichord(notes) {

      // Trigger the notes simultaneously with a duration of 1 second

      synth.triggerAttackRelease(notes, '1s');

    }

  </script>
